148  static void I2cHandler(const Bus_t & i2c)
149  {
150  MasterState state = MasterState(i2c.registers->STAT);
151  uint32_t clear_mask = 0;
152  uint32_t set_mask = 0;
153  switch (state)
154  {
155  case MasterState::kBusError: // 0x00
156  {
157  i2c.transaction.status = Status::kBusError;
158  set_mask = Control::kAssertAcknowledge | Control::kStop;
159  break;
160  }
161  case MasterState::kStartCondition: // 0x08
162  {
163  i2c.registers->DAT = i2c.transaction.GetProperAddress();
164  break;
165  }
166  case MasterState::kRepeatedStart: // 0x10
167  {
168  i2c.transaction.operation = Operation::kRead;
169  i2c.registers->DAT = i2c.transaction.GetProperAddress();
170  break;
171  }
172  case MasterState::kSlaveAddressWriteSentReceivedAck: // 0x18
173  {
174  clear_mask = Control::kStart;
175  if (i2c.transaction.out_length == 0)
176  {
177  i2c.transaction.busy = false;
178  i2c.transaction.status = Status::kSuccess;
179  set_mask = Control::kStop;
180  }
181  else
182  {
183  size_t position = i2c.transaction.position++;
184  i2c.registers->DAT = i2c.transaction.data_out[position];
185  }
186  break;
187  }
188  case MasterState::kSlaveAddressWriteSentReceivedNack: // 0x20
189  {
190  clear_mask = Control::kStart;
191  i2c.transaction.busy = false;
192  i2c.transaction.status = Status::kDeviceNotFound;
193  set_mask = Control::kStop;
194  break;
195  }
196  case MasterState::kTransmittedDataReceivedAck: // 0x28
197  {
198  if (i2c.transaction.position >= i2c.transaction.out_length)
199  {
200  if (i2c.transaction.repeated)
201  {
202  // OR with 1 to set address as READ for the next transaction
203  i2c.transaction.operation = Operation::kRead;
204  i2c.transaction.position = 0;
205  set_mask = Control::kStart;
206  }
207  else
208  {
209  i2c.transaction.busy = false;
210  set_mask = Control::kStop;
211  }
212  }
213  else
214  {
215  size_t position = i2c.transaction.position++;
216  i2c.registers->DAT = i2c.transaction.data_out[position];
217  }
218  break;
219  }
220  case MasterState::kTransmittedDataReceivedNack: // 0x30
221  {
222  i2c.transaction.busy = false;
223  set_mask = Control::kStop;
224  break;
225  }
226  case MasterState::kArbitrationLost: // 0x38
227  {
228  set_mask = Control::kStart;
229  break;
230  }
231  case MasterState::kSlaveAddressReadSentReceivedAck: // 0x40
232  {
233  clear_mask = Control::kStart;
234  if (i2c.transaction.in_length == 0)
235  {
236  set_mask = Control::kStop;
237  }
238  // If we only want 1 byte, make sure to nack that byte
239  else if (i2c.transaction.in_length == 1)
240  {
241  clear_mask |= Control::kAssertAcknowledge;
242  }
243  // If we want more then 1 byte, make sure to ack the first byte
244  else
245  {
246  set_mask = Control::kAssertAcknowledge;
247  }
248  break;
249  }
250  case MasterState::kSlaveAddressReadSentReceivedNack: // 0x48
251  {
252  clear_mask = Control::kStart;
253  i2c.transaction.status = Status::kDeviceNotFound;
254  i2c.transaction.busy = false;
255  set_mask = Control::kStop;
256  break;
257  }
258  case MasterState::kReceivedDataReceivedAck: // 0x50
259  {
260  const size_t kBufferEnd = i2c.transaction.in_length;
261  if (i2c.transaction.position < kBufferEnd)
262  {
263  const size_t kPosition = i2c.transaction.position;
264  i2c.transaction.data_in[kPosition] =
265  static_cast<uint8_t>(i2c.registers->DAT);
266  i2c.transaction.position++;
267  }
268  // Check if the position has been pushed past the buffer length
269  if (i2c.transaction.position + 1 >= kBufferEnd)
270  {
271  clear_mask = Control::kAssertAcknowledge;
272  i2c.transaction.busy = false;
273  }
274  else
275  {
276  set_mask = Control::kAssertAcknowledge;
277  }
278  break;
279  }
280  case MasterState::kReceivedDataReceivedNack: // 0x58
281  {
282  i2c.transaction.busy = false;
283  if (i2c.transaction.in_length != 0)
284  {
285  size_t position = i2c.transaction.position++;
286  i2c.transaction.data_in[position] =
287  static_cast<uint8_t>(i2c.registers->DAT);
288  }
289  set_mask = Control::kStop;
290  break;
291  }
292  case MasterState::kDoNothing: // 0xF8
293  {
294  break;
295  }
296  default:
297  {
298  clear_mask = Control::kStop;
299  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
300  break;
301  }
302  }
303  // Clear I2C Interrupt flag
304  clear_mask |= Control::kInterrupt;
305  // Set register controls
306  i2c.registers->CONSET = set_mask;
307  i2c.registers->CONCLR = clear_mask;
308  }

333  // Calculating and setting the I2C Clock rate
334  // Weight the high side duty cycle more than the lower side by 30% in order
335  // to give more time for the bus to charge up.
336  float clock_frequency = (peripheral_frequency / clock_rate_) / 2.0f;
337  float scll = clock_frequency * 0.7f;
338  float sclh = clock_frequency * 1.3f;
339  i2c_.registers->SCLL = static_cast<uint32_t>(scll);
340  i2c_.registers->SCLH = static_cast<uint32_t>(sclh);
341

343  Control::kStop | Control::kInterrupt;
344  i2c_.registers->CONSET = Control::kInterfaceEnable;

379  {
380  // Skip waiting on the interrupt if running a host unit test
381  if constexpr (build::kPlatform == build::Platform::host)
382  {
383  return {};
384  }
385 
386  if (!IsIntialized())

394  return !i2c_.transaction.busy;
395  };
396 
397  Status wait_status =
398  Wait(i2c_.transaction.timeout, wait_for_i2c_transaction);
399 
400  if (i2c_.transaction.status == Status::kBusError)
401  {
402  return CommonErrors::kBusError;
403  }
404  else if (i2c_.transaction.status == Status::kDeviceNotFound)
405  {
406  return CommonErrors::kDeviceNotFound;
407  }
408  else if (wait_status == Status::kTimedOut)
409  {
410  // Abort I2C communication if this point is reached!
411  i2c_.registers->CONSET = Control::kAssertAcknowledge | Control::kStop;
412 
413  if (i2c_.transaction.out_length == 0 || i2c_.transaction.in_length == 0)
414  {
415  return CommonErrors::kTimeout;
416  }
417  }
418 
419  // Ensure that start is cleared before leaving this function
420  i2c_.registers->CONCLR = Control::kStart;
421  return {};
