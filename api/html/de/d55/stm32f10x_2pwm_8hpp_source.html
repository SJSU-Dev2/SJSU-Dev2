213  void ConfigureFrequency()
214  {
215  static constexpr auto kUpdateGeneration = bit::MaskFromRange(0);
216 
217  auto & system = SystemController::GetPlatformController();
218  const auto kPeripheralFrequency = system.GetClockRate(channel_.id);
219 
220  auto period = 1 / settings.frequency;
221  uint32_t product = kPeripheralFrequency * period;
222 
223  channel_.registers->PSC = GetPrescalarValue(product);
224  channel_.registers->ARR =
225  static_cast<uint16_t>(product / (channel_.registers->PSC + 1));
226 
227  bit::Register(&channel_.registers->EGR).Set(kUpdateGeneration).Save();
228  }

230  uint16_t GetPrescalarValue(uint32_t product)
231  {
232  uint16_t prescalar = 0;
233  uint16_t k_max16_bits = ~0;
234 
235  uint32_t arr;
236 
237  // To select a prescalar and divider value, we use this equation:
238  // (PSC+1)(ARR+1) = (EventTime)(ClkFreq)
239  // product is the value (EventTime)(ClkFreq)
240  //
241  // We want ARR to be a larger value for a more precise duty cycle
242  // so we increment prescalar until ARR in the value (PSC+1)(ARR+1)
243  // is less than 2^16
244  do
245  {
246  prescalar++;
247  arr = product / prescalar;
248  } while (arr > k_max16_bits);
249 
250  return --prescalar;
251  }
