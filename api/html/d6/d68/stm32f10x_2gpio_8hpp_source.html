80  static void InterruptHandler()
81  {
82  // Find the pin that triggered this interrupt by reading the Pending
83  // Register. Each bit location number represents the pin number that
84  // triggered this interrupt.
85  // We can count the number of zeros from the right to return the bit number
86  // of the least significant set bit in the register, which will be at least
87  // one of the pins that triggered this handler.
88  int pin = __builtin_ctz(external_interrupt->PR);
89 
90  // Use the pin to lookup and run the associated callback
91  handlers[pin]();
92 
93  // Clear the pending interrupt by writing a 1 back to the PR register.
94  external_interrupt->PR = (1 << pin);
95  }

97  void AttachInterrupt(InterruptCallback callback, Edge edge) override
98  {
99  // Clear both falling and raising edges bits
100  // They will be assigned in the conditionals below
101  DetachInterrupt();
102 
103  // Add callback to list of handlers
104  handlers[pin_.GetPin()] = callback;
105 
106  if (edge == Edge::kBoth || edge == Edge::kRising)
107  {
108  external_interrupt->RTSR =
109  bit::Set(external_interrupt->RTSR, pin_.GetPin());
110  }
111  if (edge == Edge::kBoth || edge == Edge::kFalling)
112  {
113  external_interrupt->FTSR =
114  bit::Set(external_interrupt->FTSR, pin_.GetPin());
115  }
116 
117  // Enable interrupts for this particular pin
118  external_interrupt->IMR = bit::Set(external_interrupt->IMR, pin_.GetPin());
119 
120  // Fetch the external interrupt control register from the AFIO peripheral
121  // There are 4 EXTICR registers which only use the first 16 bits of the
122  // 32-bit register.
123  // We have 16 EXTI interrupt signals which can be assigned to any of the
124  // ports A, B, ..., G.
125  // Divide the pin number by 4 to select the EXTICR register in the array
126  // to update.
127  volatile uint32_t * control = &Pin::afio->EXTICR[pin_.GetPin() / 4];
128 
129  // The location within the EXTICR[x] register for the port select
130  // information
131  auto interrupt_alternative_function_mask = bit::Mask{
132  .position = static_cast<uint32_t>((pin_.GetPin() * 4) % 16),
133  .width = 4,
134  };
135 
136  // Assign the control port value, which is equal to the port value minus
137  // 'A', thus A would be 0, B would be 1, etc.
138  *control = bit::Insert(
139  *control, pin_.GetPort() - 'A', interrupt_alternative_function_mask);
140 
141  // Enable interrupts for this external interrupt line
142  switch (pin_.GetPin())
143  {
144  case 0:

203  void DetachInterrupt() override
204  {

206  // especially since you would need logic to determine if this is the last
