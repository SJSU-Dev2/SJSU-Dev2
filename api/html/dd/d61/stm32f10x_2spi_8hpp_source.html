177  template <typename T>
178  void UniversalTransfer(std::span<T> buffer)
179  {
180  // Follows the instructions provided by RM0008.pdf page 710 to get the
181  // maximum performance.
182 
183  // Step 1. Load the first byte into the data register
184  port_.spi->DR = buffer[0];
185 
186  for (size_t i = 1; i < buffer.size(); i++)
187  {
188  // Step 2.1 Wait until the transmit buffer is empty, then...
189  while (!TransmitBufferEmpty())
190  {
191  continue;
192  }
193 
194  // Step 2.2 Load it with the next byte.
195  port_.spi->DR = buffer[i];
196 
197  // Step 2.3 Wait until the receive buffer has data.
198  while (!ReceiveBufferNotEmpty())
199  {
200  continue;
201  }
202 
203  // Step 2.4 Read byte from DR register and place it into the previous byte
204  // location. This is important as the bytes written are head of the bytes
205  // received by 1.
206  buffer[i - 1] = static_cast<T>(port_.spi->DR);
207  }
208 
209  // Step 3.1 Wait for the last byte to come in on the receive buffer.
210  while (!ReceiveBufferNotEmpty())
211  {
212  continue;
213  }
214 
215  // Step 3.2 Read the final byte from the receive buffer and place it in the
216  // end of the span.
217  buffer.end()[-1] = static_cast<T>(port_.spi->DR);
218  }
