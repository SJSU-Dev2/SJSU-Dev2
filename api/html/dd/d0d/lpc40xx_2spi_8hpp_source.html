214  void ConfigureFrequency()
215  {
216  auto & system = sjsu::SystemController::GetPlatformController();
217  auto system_frequency = system.GetClockRate(bus_.power_on_bit);
218 
219  auto prescaler = system_frequency / settings.clock_rate;
220 
221  // Store lower half of prescalar in clock prescalar register
222  bus_.registers->CPSR = prescaler.to<uint16_t>() & 0xFF;
223 
224  // Store upper 8 bit half of the prescalar in control register 0
225  bus_.registers->CR0 = bit::Insert(bus_.registers->CR0,
226  prescaler.to<uint16_t>() >> 8,
227  ControlRegister0::kDividerBit);
228  }

241  void ConfigureFrameSize()
242  {

245  // to get the appropriate tranfer code.
