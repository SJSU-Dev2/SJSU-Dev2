294 
295  uint16_t prescaler = (system_frequency / frequency).to<uint16_t>();
296  // Store lower half of prescalar in clock prescalar register
297  bus_.registers->CPSR = prescaler & 0xFF;
298  // Store upper 8 bit half of the prescalar in control register 0
299  bus_.registers->CR0 = bit::Insert(bus_.registers->CR0, prescaler >> 8,
