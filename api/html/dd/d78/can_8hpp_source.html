230  Can::Message_t SecureGet()
231  {
232  // Continuously check if the received CAN message is valid. NOTE: that, in
233  // general, the looping logic for this function almost never occurs as
234  // copies from RAM/cache are fast between ReceiveHandler and GetMessage()
235  while (true)
236  {
237  // First, atomically copy the access_counter to a variable.
238  const auto kReadMessageStart = access_counter.load();
239 
240  // Copy the contents of the mesage into the kCanMessage variable.
241  const Can::Message_t kCanMessage = data;
242 
243  // Finally, atomically copy the access_counter again such that the value
244  // between the start and finish can be compared.
245  const auto kReadMessageFinish = access_counter.load();
246 
247  // If and only if these two values are the same, meaning that the
248  // ReceiveHandler isn't currently modifying the Node_t's data,
249  // can the message be considered valid.
250  if (kReadMessageStart == kReadMessageFinish)
251  {
252  return kCanMessage;
253  }
254  }
255 
256  return {};
257  };
258 
259  private:
260  friend CanNetwork;
261 
266  void Update(const Can::Message_t & new_data)
267  {
268  // Atomic increment of the access counter to notify any threads that are
269  // using GetMesage() that the value of this node is changing.
270  access_counter++;
271 
272  // Copy the contents of the new message into the map.
273  data = new_data;
274 
275  // Another atomic increment of the access counter to notify any thread
276  // using GetMessage() that the value of this node has finished changing.
277  // For GetMessage to be valid, the function must save the access_counter,
278  // copy the message data, then check the access_counter again to make sure
279  // it has not changed. If it has changed, then GetMessage() should
280  // continue polling the map node until access_counter is the same between
281  // copies.
282  access_counter++;
283  }

374  void ReceiveHandler(sjsu::Can & can)
375  {
376  // If there isn't any data available, return early.
377  if (!can.HasData())
378  {
379  return;
380  }
381 
382  // Pop the latest can message off the queue.
383  const auto kMessage = can.Receive();
384 
385  // Check if the map already has a value for this ID. This acts as the last
386  // stage of the CAN filter for the CANBUS Network module. If the key
387  // does NOT exist in the map, then this message will not be saved.
388  // Typically, this only happens when the hardware filter has not been setup
389  // properly to eliminate can messages that should be ignored.
390  //
391  // Map lookups can be costly, especially in a interrupt context, so only
392  // needing to hash/lookup the ID once is preferred. To prevent multiple
393  // lookups, an iterator is stored into the message_node variable.
394  auto message_node = messages_.find(kMessage.id);
395 
396  // If the ID has an associated value in the map, then the node std::pair<>
397  // is returned, otherwise, the ::end() node is returned.
398  if (message_node != messages_.end())
399  {
400  message_node->second.Update(kMessage);
401  }
402  }
