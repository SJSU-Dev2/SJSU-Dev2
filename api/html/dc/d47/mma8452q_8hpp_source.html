94  Acceleration_t Read() override
95  {
96  Acceleration_t acceleration = {};
97  std::array<int16_t, 3> result = memory_[Map::kXYZStartAddress];
98  auto xyz_data = result;
99 
100  // First X-axis Byte (MSB first)
101  // =========================================================================
102  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
103  // XD11 | XD10 | XD9 | XD8 | XD7 | XD6 | XD5 | XD4
104  //
105  // Final X-axis Byte (LSB)
106  // =========================================================================
107  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
108  // XD3 | XD2 | XD1 | XD0 | 0 | 0 | 0 | 0
109  //
110  // We simply shift and OR the bytes together to get them into a signed int
111  // 16 value. We do not shift yet because we want to get the signed bit in
112  // the most significant bit position to allow for sign extension when we
113  // shift to the right later.
114 
115  int16_t x = static_cast<int16_t>(xyz_data[0] >> 4);
116  int16_t y = static_cast<int16_t>(xyz_data[1] >> 4);
117  int16_t z = static_cast<int16_t>(xyz_data[2] >> 4);
118 
119  // Convert the 12-bit signed value into a value from -1.0 to 1.0f so it can
120  // be multiplied by the full_scale in order to get the true acceleration.
121  constexpr int16_t kMin = sjsu::BitLimits<12, int16_t>::Min();
122  constexpr int16_t kMax = sjsu::BitLimits<12, int16_t>::Max();
123 
124  float x_axis_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
125  float y_axis_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
126  float z_axis_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
127 
128  acceleration.x = CurrentSettings().gravity * x_axis_ratio;
129  acceleration.y = CurrentSettings().gravity * y_axis_ratio;
130  acceleration.z = CurrentSettings().gravity * z_axis_ratio;
131 
132  return acceleration;
133  }
