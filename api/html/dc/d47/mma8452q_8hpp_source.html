81  Returns<Acceleration_t> Read() override
82  {
83  constexpr uint16_t kBytesPerAxis = 2;
84  constexpr uint8_t kNumberOfAxis = 3;
85 
86  Acceleration_t acceleration = {};
87 
88  uint8_t xyz_data[kBytesPerAxis * kNumberOfAxis];
89 
90  SJ2_RETURN_ON_ERROR(i2c_.WriteThenRead(
91  kAccelerometerAddress, { Value(RegisterMap::kXYZStartAddress) },
92  xyz_data, sizeof(xyz_data)));
93 
94  // First X-axis Byte (MSB first)
95  // =========================================================================
96  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
97  // XD11 | XD10 | XD9 | XD8 | XD7 | XD6 | XD5 | XD4
98  //
99  // Final X-axis Byte (LSB)
100  // =========================================================================
101  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
102  // XD3 | XD2 | XD1 | XD0 | 0 | 0 | 0 | 0
103  //
104  // We simply shift and OR the bytes together to get them into a signed int
105  // 16 value. We do not shift yet because we want to get the signed bit in
106  // the most significant bit position to allow for sign extension when we
107  // shift to the right later.
108  int16_t x = static_cast<int16_t>(xyz_data[0] << 8 | xyz_data[1]);
109  int16_t y = static_cast<int16_t>(xyz_data[2] << 8 | xyz_data[3]);
110  int16_t z = static_cast<int16_t>(xyz_data[4] << 8 | xyz_data[5]);
111 
112  // Each axis is left shifted by 4 bits in order to place the signed bit in
113  // the 16th bit position for the 16 bit integer. This allows us to right
114  // shift to get rid of the right most zeros and to sign extend the value,
115  // allowing us to return the correct value of the axis.
116  x = static_cast<int16_t>(x >> 4);
117  y = static_cast<int16_t>(y >> 4);
118  z = static_cast<int16_t>(z >> 4);
119 
120  // Convert the 12-bit signed value into a value from -1.0 to 1.0f so it can
121  // be multiplied by the kFullScale in order to get the true acceleration.
122  constexpr int16_t kMin = sjsu::BitLimits<12, int16_t>::Min();
123  constexpr int16_t kMax = sjsu::BitLimits<12, int16_t>::Max();
124 
125  float x_axis_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
126  float y_axis_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
127  float z_axis_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
128 
129  sjsu::LogDebug("x = %d :: y = %d :: z = %d", x, y, z);
130 
131  acceleration.x = kFullScale * x_axis_ratio;
132  acceleration.y = kFullScale * y_axis_ratio;
133  acceleration.z = kFullScale * z_axis_ratio;
134 
135  return acceleration;
136  }
