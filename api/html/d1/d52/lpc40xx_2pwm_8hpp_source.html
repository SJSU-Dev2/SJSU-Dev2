121  void ModuleInitialize() override
122  {
123  auto & system = sjsu::SystemController::GetPlatformController();
124  system.PowerUpPeripheral(channel_.peripheral.id);
125 
126  // Get a pointer to the PWM registers
127  auto * pwm = channel_.peripheral.registers;
128 
129  // Set prescalar to 1 so the input frequency to the PWM peripheral is
130  // equal to the peripheral clock frequency.
131  //
132  // PR = Prescale register = defines the maximum value for the prescaler
133  pwm->PR = 0;
134 
135  // PC = Prescale counter. Set to 0 to cause the timer counter to increment
136  // after each peripheral clock tick.
137  pwm->PC = 0;
138 
139  // Mode 0x0 means increment time counter (TC) after the peripheral clock
140  // has cycled the amount inside of the prescale.
141  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kMode);
142 
143  // 0x0 for this register says to use the TC for input counts.
144  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kCountInput);
145 
146  // Sets match register 0 to reset when TC and Match 0 match each other,
147  // meaning that the PWM pulse will cycle continuously.
148  pwm->MCR = bit::Set(pwm->MCR, MatchControl::kPwm0Reset);
149 
150  // Enables PWM[channel] output
151  const uint32_t kOutputEnableStart = OutputControl::kEnableOutput.position;
152  const uint32_t kChannelBitPosition = kOutputEnableStart + channel_.channel;
153  pwm->PCR = bit::Set(pwm->PCR, kChannelBitPosition);
154 
155  // Set the pin function
156  channel_.pin.settings.function = channel_.pin_function_code;
157  channel_.pin.Initialize();
158 
159  // Set PWM frequency
160  ConfigureFrequency();
161 
162  // Enable the PWM output channel
163  Enable();
164  }

171  void SetDutyCycle(float duty_cycle) override
172  {
173  // Clamp duty cycle
174  const float kClampedDuty = std::clamp(duty_cycle, 0.0f, 1.0f);
175 
176  // Set match register for this channel
177  GetMatchRegisters(channel_.channel) = CalculateDutyCycle(kClampedDuty);
178 
179  // LER (Load Enable Register) sets the hardware up to load the new duty
180  // cycle period in the next reset cycle after MR0 match. Hardware requires
181  // this as a means ot prevent itself from glitching and changing pwm
182  // instantaniously.
183  channel_.peripheral.registers->LER |= (1 << channel_.channel);
184  }

186  float GetDutyCycle() override
187  {
188  auto period = static_cast<float>(GetMatchRegisters(channel_.channel));
189  auto max_period = static_cast<float>(GetMatchRegisters(0));
190  return period / max_period;

194  void ConfigureFrequency()
195  {
196  if (settings.frequency <= 0_Hz)
197  {
198  SJ2_PRINT_VARIABLE(settings.frequency, "%f");
199  throw Exception(
200  std::errc::invalid_argument,
201  "Frequency must be greater than 0 Hz. Frequency will not be "
202  "updated.");
203  }
204 
205  Enable(false);
206 
207  auto & system = sjsu::SystemController::GetPlatformController();
208  auto peripheral_frequency = system.GetClockRate(channel_.peripheral.id);
209 
210  // Get the current duty cycle so we can match it to the updated frequency.
211  float previous_duty_cycle = GetDutyCycle();
212 
213  // In order to avoid PWM glitches, the PWM must be disabled while updating
214  // the MR0 register. Doing this will reset all counters to 0 and allow us to
215  // update MR0.
216  const auto kNewFrequency = peripheral_frequency / settings.frequency;
217  GetMatchRegisters(0) = kNewFrequency.to<uint32_t>();
218 
219  // Re-enable PWM which will also reset all of the counters which allow
220  // setting the duty cycle of other PWM channels.
221  Enable(true);
222 
223  SetDutyCycle(previous_duty_cycle);
224  }

226  void Enable(bool enable = true)
227  {
228  bit::Register pwm_timer(&channel_.peripheral.registers->TCR);
229 
230  if (enable)
231  {
232  // Reset the Timer Counter
233  pwm_timer.Set(Timer::kCounterReset).Save();
234 
235  // Clear reset and allow timer to count
236  pwm_timer.Clear(Timer::kCounterReset).Save();
237 
238  // Enable PWM output
239  pwm_timer.Set(Timer::kPwmEnable).Save();
240 
241  // Enable counting
242  pwm_timer.Set(Timer::kCounterEnable).Save();
243  }
244  else
245  {
246  // Disable PWM output
247  pwm_timer.Clear(Timer::kPwmEnable).Save();
248  }
