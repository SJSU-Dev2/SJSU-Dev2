194  Status Initialize(units::frequency::hertz_t frequency_hz) const override
195  {
196  SJ2_ASSERT_FATAL(1 <= channel_.channel && channel_.channel <= 6,
197  "Channel must be between 1 and 6 on LPC40xx platforms.");
198 
199  auto & system = sjsu::SystemController::GetPlatformController();
200 
201  system.PowerUpPeripheral(channel_.peripheral.id);
202 
203  auto * pwm = channel_.peripheral.registers;
204  // Set prescalar to 1 so the input frequency to the PWM peripheral is equal
205  // to the peripheral clock frequency.
206  //
207  // PR = Prescale register = defines the maximum value for the prescaler
208  pwm->PR = 0;
209 
210  // PC = Prescale counter. Set to 0 to cause the timer counter to increment
211  // after each peripheral clock tick.
212  pwm->PC = 0;
213 
214  // Mode 0x0 means increment time counter (TC) after the peripheral clock
215  // has cycled the amount inside of the prescale.
216  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kMode);
217 
218  // 0x0 for this register says to use the TC for input counts.
219  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kCountInput);
220 
221  // Match register 0 is used to generate the desired frequency. If the time
222  // counter TC is equal to MR0
223  const units::frequency::hertz_t kPeripheralFrequency =
224  system.GetClockRate(channel_.peripheral.id);
225 
226  pwm->MR0 = (kPeripheralFrequency / frequency_hz).to<uint32_t>();
227 
228  // Sets match register 0 to reset when TC and Match 0 match each other,
229  // meaning that the PWM pulse will cycle continuously.
230  pwm->MCR = bit::Set(pwm->MCR, MatchControl::kPwm0Reset);
231 
232  // Enables PWM TC and PC for counting and enables PWM mode
233  EnablePwm();
234 
235  // Enables PWM[channel] output
236  uint32_t channel_bit_position =
237  OutputControl::kEnableOutput.position + channel_.channel;
238  pwm->PCR = bit::Set(pwm->PCR, channel_bit_position);
239 
240  channel_.pin.SetPinFunction(channel_.pin_function_code);
241 
242  return Status::kSuccess;
243  }

245  void SetDutyCycle(float duty_cycle) const override
246  {
247  SJ2_ASSERT_FATAL(0.0f <= duty_cycle && duty_cycle <= 1.0f,
248  "duty_cycle of Duty Cycle provided is out of bounds.");
249  *GetMatchRegisters(channel_.channel) = CalculateDutyCycle(duty_cycle);
250  // LER (Load Enable Register) sets the hardware up to load the new duty
251  // cycle period in the next reset cycle after MR0 match. Hardware requires
252  // this as a means ot prevent itself from glitching and changing pwm
253  // instantaniously.
254  channel_.peripheral.registers->LER |= (1 << channel_.channel);
255  }

265  {
266  SJ2_RETURN_IF(

269

271 
272  auto peripheral_frequency = system.GetClockRate(channel_.peripheral.id);
273  // Get the current duty cycle so we can match it to the updated frequency.
274  float previous_duty_cycle = GetDutyCycle();
275 
276  // In order to avoid PWM glitches, the PWM must be disabled while updating
277  // the MR0 register. Doing this will reset all counters to 0 and allow us to
278  // update MR0.
279  EnablePwm(false);
280  {
281  auto new_frequency = (peripheral_frequency / frequency_hz).to<uint32_t>();
282  *GetMatchRegisters(0) = new_frequency;
283  SetDutyCycle(previous_duty_cycle);
284  }

309  void EnablePwm(bool enable = true) const
310  {
311  volatile uint32_t * pwm_timer = &channel_.peripheral.registers->TCR;
312  if (enable)
