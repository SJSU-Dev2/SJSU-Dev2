265  {
266  auto control_register1 = bit::Register(&port_.uart->CR1);
267  constexpr auto kParityControl = bit::MaskFromRange(10);
268  constexpr auto kParitySelection = bit::MaskFromRange(9);
269  constexpr auto kWordLength = bit::MaskFromRange(12);
270 
271  auto control_register2 = bit::Register(&port_.uart->CR2);
272  constexpr auto kStop = bit::MaskFromRange(12, 13);
273 
274  if (settings.frame_size < UartSettings_t::FrameSize::kEightBits)
275  {
276  throw Exception(std::errc::not_supported,
277  "Only 8-bit and 9-bit frame sizes are supported!");
278  }
279 
280  bool double_stop = (settings.stop == UartSettings_t::StopBits::kDouble);
281  uint16_t stop_value = (double_stop) ? 0b10 : 0b00;
282  bool parity_enable = (settings.parity != UartSettings_t::Parity::kNone);
283  bool parity = (settings.parity == UartSettings_t::Parity::kOdd);
284  bool size_code =
285  (settings.frame_size == UartSettings_t::FrameSize::kNineBits);
286  // Parity codes are: 0 for Even and 1 for Odd, thus the expression above
287  // sets the bool to TRUE when odd and zero when something else. This value
288  // is ignored if the parity is NONE since parity_enable will be zero.
289 
290  control_register1.Insert(parity_enable, kParityControl)
291  .Insert(parity, kParitySelection)
292  .Insert(size_code, kWordLength)
293  .Save();

298  void ConfigureBaudRate()
299  {
300  auto & system = SystemController::GetPlatformController();
301  auto frequency = system.GetClockRate(port_.id);
302 
303  float float_baud_rate = static_cast<float>(settings.baud_rate);
304  float usart_divider =
305  static_cast<float>(frequency) / (16.0f * float_baud_rate);
306 
307  // Truncate off the decimal values
308  uint16_t mantissa = static_cast<uint16_t>(usart_divider);
309 
310  // Subtract the whole number to leave just the decimal
311  float fraction = static_cast<float>(usart_divider - mantissa);
312 
313  uint16_t fractional_int = static_cast<uint16_t>(std::roundf(fraction * 16));
314 
315  if (fractional_int >= 16)
316  {
317  mantissa = static_cast<uint16_t>(mantissa + 1U);
318  fractional_int = 0;
319  }
320 
321  port_.uart->BRR = bit::Value<uint16_t>()
322  .Insert(mantissa, BaudRateReg::kMantissa)
323  .Insert(fractional_int, BaudRateReg::kFraction);
324  }
