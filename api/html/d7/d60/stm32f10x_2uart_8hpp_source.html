307  void SetBaudRate(uint32_t baud_rate) const override
308  {
309  auto & system = SystemController::GetPlatformController();
310  auto frequency = system.GetClockRate(port_.id);
311 
312  float float_baud_rate = static_cast<float>(baud_rate);
313  float usart_divider =
314  static_cast<float>(frequency) / (16.0f * float_baud_rate);
315 
316  // Truncate off the decimal values
317  uint16_t mantissa = static_cast<uint16_t>(usart_divider);
318 
319  // Subtract the whole number to leave just the decimal
320  float fraction = static_cast<float>(usart_divider - mantissa);
321 
322  uint16_t fractional_int = static_cast<uint16_t>(std::roundf(fraction * 16));
323 
324  if (fractional_int >= 16)

326  mantissa = static_cast<uint16_t>(mantissa + 1U);
327  fractional_int = 0;
328  }
329 
330  port_.uart->BRR = bit::Value<uint16_t>()
331  .Insert(mantissa, BaudRateReg::kMantissa)
332  .Insert(fractional_int, BaudRateReg::kFraction);
